    version: 0.1
    backend:
      phases:
        # IMPORTANT - Please verify your build commands
        build:
          commands:
            - '# Execute Amplify CLI with the helper script'
            - amplifyPush --simple
    frontend:
      phases:
        preBuild:  # install the files to build
          commands:
            - ls -al
            - cd frontend-store
            - npm install
        build:
          commands:
            - ls -al
            - npm run build
      artifacts:
        # IMPORTANT - Please verify your build output directory
        baseDirectory: frontend-store/dist
        files:
          - '**/*'
      cache:
        paths:
          - frontend-store/node_modules
